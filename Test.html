<button onclick="getSpeed()">Check my speed</button>


<script>

function getSpeed() {

if (navigator.geolocation) {

navigator.geolocation.watchPosition(function(position) {

// `position.coords.speed` will give the speed in meters per second

// Convert to km/h by multiplying with 3.6

alert("Your speed is " + (position.coords.speed * 3.6).toFixed(2) + " km/h");

});

} else {

alert("Geolocation is not supported by this browser.");

}

}

</script>
